## Observables in Xania

Xania support any observable that satisfies the `Observable` interface

```javascript
/** Observable pattern support in @xania/view */

interface Observable<T> {
  subscribe<TObservable extends NextObserver<T>>(observer: TNextObservable): Unsubscribable;
}

interface NextObserver<T> {
  next(value: T): void;
}
```

```javascript
function Counter() {
  export function UseStateDemo() {
    const count = useState(0);

    return (
      <div class="element">
        <div>Count: {count}</div>
        <button class="mdc-button" click={_ => count.update(count.value + 1)}>
          <span>Increment</span>
        </button>
      </div>
    );
  }
}
```
